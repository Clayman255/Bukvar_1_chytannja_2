    <script>
        const data = [
            { id: 1, img: '1.png', word: 'прудкий', aImg: '01.mp3', aWord: 'au01.mp3' },
            { id: 2, img: '2.png', word: 'ряба', aImg: '02.mp3', aWord: 'au02.mp3' },
            { id: 3, img: '3.png', word: 'маленьке', aImg: '03.mp3', aWord: 'au03.mp3' },
            { id: 4, img: '4.png', word: 'замурзані', aImg: '04.mp3', aWord: 'au04.mp3' },
            { id: 5, img: '5.png', word: 'вайлувата', aImg: '02.mp3', aWord: 'au05.mp3' },
            { id: 6, img: '6.png', word: 'пухнастий', aImg: '01.mp3', aWord: 'au06.mp3' }
        ];

        let selectedWordId = null;
        let matchedCount = 0;
        let connections = [];

        const canvas = document.getElementById('line-canvas');
        const ctx = canvas.getContext('2d');

        function playSound(file) {
            new Audio(file).play().catch(() => {});
        }

        window.onload = () => {
            initGame();
            resizeCanvas();
            setTimeout(() => playSound('zav.mp3'), 500);
        };

        window.onresize = resizeCanvas;
        window.onscroll = drawArrows;

        function resizeCanvas() {
            canvas.width = document.documentElement.scrollWidth;
            canvas.height = document.documentElement.scrollHeight;
            drawArrows();
        }

        function initGame() {
            const leftCol = document.getElementById('left-targets');
            const rightCol = document.getElementById('right-targets');
            const wordsCol = document.getElementById('words-container');

            data.forEach((item, index) => {
                const zone = document.createElement('div');
                zone.className = 'drop-zone';
                zone.id = `target-${item.id}`;
                zone.onclick = () => handleTargetClick(item.id);
                zone.innerHTML = `
                    <img src="${item.img}" class="animal-img">
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.aImg}')">🔊</button>
                `;
                if (index < 3) leftCol.appendChild(zone);
                else rightCol.appendChild(zone);
            });

            const shuffled = [...data].sort(() => Math.random() - 0.5);
            shuffled.forEach(item => {
                const card = document.createElement('div');
                card.className = 'word-card';
                card.id = `word-${item.id}`;
                card.onclick = () => handleWordClick(item.id);
                card.innerHTML = `
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.aWord}')">🔊</button>
                    <span>${item.word}</span>
                `;
                wordsCol.appendChild(card);
            });
        }

        function handleWordClick(id) {
            if (document.getElementById(`word-${id}`).classList.contains('matched')) return;
            document.querySelectorAll('.word-card').forEach(c => c.classList.remove('selected'));
            selectedWordId = id;
            document.getElementById(`word-${id}`).classList.add('selected');
        }

        function handleTargetClick(targetId) {
            if (!selectedWordId) return;

            if (selectedWordId === targetId) {
                const wordEl = document.getElementById(`word-${targetId}`);
                const targetEl = document.getElementById(`target-${targetId}`);
                
                wordEl.classList.remove('selected');
                wordEl.classList.add('matched');
                targetEl.classList.add('correct');
                
                connections.push({ wordId: targetId, targetId: targetId });
                selectedWordId = null;
                matchedCount++;
                drawArrows();

                if (matchedCount === data.length) {
                    document.getElementById('win-text').style.display = 'block';
                }
            }
        }

        function drawArrows() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#3498db';
            ctx.lineWidth = 5;
            ctx.lineCap = 'round';

            connections.forEach(conn => {
                const wEl = document.getElementById(`word-${conn.wordId}`);
                const tEl = document.getElementById(`target-${conn.targetId}`);
                
                const wRect = wEl.getBoundingClientRect();
                const tRect = tEl.getBoundingClientRect();

                // Центр картинки
                const targetX = tRect.left + tRect.width / 2 + window.scrollX;
                const targetY = tRect.top + tRect.height / 2 + window.scrollY;

                // Визначаємо край слова (лівий чи правий) залежно від розташування картинки
                let wordX;
                if (targetX < (wRect.left + window.scrollX)) {
                    // Картинка зліва від слова -> тягнемо до лівої грані слова
                    wordX = wRect.left + window.scrollX;
                } else {
                    // Картинка справа від слова -> тягнемо до правої грані слова
                    wordX = wRect.right + window.scrollX;
                }
                const wordY = wRect.top + wRect.height / 2 + window.scrollY;

                ctx.beginPath();
                ctx.moveTo(targetX, targetY);
                ctx.lineTo(wordX, wordY);
                ctx.stroke();

                // Точки на кінцях
                ctx.fillStyle = '#3498db';
                ctx.beginPath();
                ctx.arc(targetX, targetY, 6, 0, Math.PI * 2);
                ctx.arc(wordX, wordY, 6, 0, Math.PI * 2);
                ctx.fill();
            });
        }
