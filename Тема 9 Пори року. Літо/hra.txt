<script>
  const ids = [1, 2, 3, 4, 5];
  const canvas = document.getElementById('canvas');
  const ctx = canvas.getContext('2d');
  const imagesRow = document.getElementById('imagesRow');
  const audioPlayer = document.getElementById('audioPlayer');
  
  let selectedStart = null;
  let connections = []; 
  let correctCount = 0;

  function init() {
    imagesRow.innerHTML = '';
    const shuffled = [...ids].sort(() => Math.random() - 0.5);
    
    shuffled.forEach(id => {
      const wrapper = document.createElement('div');
      wrapper.className = 'item-wrapper';
      wrapper.innerHTML = `
        <div class="dot" id="end-${id}"></div>
        <div class="img-box" id="box-${id}" onclick="handleImgClick(${id})">
          <img src="${id}.png" alt="">
        </div>
      `;
      imagesRow.appendChild(wrapper);
    });

    // Оновлюємо розмір після рендерингу
    setTimeout(resizeCanvas, 300);
    window.addEventListener('resize', resizeCanvas);
  }

  function resizeCanvas() {
    const area = document.getElementById('gameArea');
    const rect = area.getBoundingClientRect();
    canvas.width = rect.width;
    canvas.height = rect.height;
    drawLines();
  }

  function playSnd(file) {
    audioPlayer.src = file;
    audioPlayer.play().catch(()=>{});
  }

  function handleSndClick(id, file) {
    playSnd(file);
    if(selectedStart) document.getElementById(`start-${selectedStart}`).classList.remove('active');
    selectedStart = id;
    document.getElementById(`start-${selectedStart}`).classList.add('active');
  }

  function handleImgClick(imgId) {
    if (!selectedStart) return;
    
    if (selectedStart === imgId) {
      if (!connections.find(c => c.id === imgId)) {
        connections.push({id: imgId});
        document.getElementById('popSnd').play();
        correctCount++;
        
        document.getElementById(`start-${imgId}`).classList.add('active');
        document.getElementById(`end-${imgId}`).classList.add('active');
        document.getElementById(`box-${imgId}`).classList.add('correct-item');
        
        selectedStart = null;
        drawLines();
        checkWin();
      }
    } else {
      const el = document.getElementById(`box-${imgId}`);
      el.style.borderColor = 'red';
      setTimeout(() => el.style.borderColor = 'transparent', 500);
    }
  }

  function drawLines() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.setLineDash([8, 6]); 
    ctx.lineWidth = 5;
    ctx.strokeStyle = '#4dabf7';

    const areaRect = document.getElementById('gameArea').getBoundingClientRect();

    connections.forEach(conn => {
      const sDot = document.getElementById(`start-${conn.id}`);
      const eDot = document.getElementById(`end-${conn.id}`);
      
      const sRect = sDot.getBoundingClientRect();
      const eRect = eDot.getBoundingClientRect();

      const x1 = sRect.left - areaRect.left + (sRect.width / 2);
      const y1 = sRect.top - areaRect.top + (sRect.height / 2);
      const x2 = eRect.left - areaRect.left + (eRect.width / 2);
      const y2 = eRect.top - areaRect.top + (eRect.height / 2);

      ctx.beginPath();
      ctx.moveTo(x1, y1);
      ctx.lineTo(x2, y2);
      ctx.stroke();
      
      const angle = Math.atan2(y2 - y1, x2 - x1);
      ctx.beginPath();
      ctx.moveTo(x2, y2);
      ctx.lineTo(x2 - 15 * Math.cos(angle - Math.PI / 7), y2 - 15 * Math.sin(angle - Math.PI / 7));
      ctx.moveTo(x2, y2);
      ctx.lineTo(x2 - 15 * Math.cos(angle + Math.PI / 7), y2 - 15 * Math.sin(angle + Math.PI / 7));
      ctx.stroke();
    });
  }

  function checkWin() {
    if (correctCount === ids.length) {
      setTimeout(() => {
        document.getElementById('winMsg').style.display = 'block';
        document.getElementById('btnTopRestart').style.display = 'block';
      }, 500);
    }
  }

  init();
</script>
