<script>
  // Дані загадок
  const riddlesData = [
    { id: 1, audioId: "z1", text: "По полю гасає,\nОвечок хапає\nТа всіх лякає.", answerImg: "im1.png" },
    { id: 2, audioId: "z2", text: "Тварина рогата,\nІ рогів багато.", answerImg: "im2.png" },
    { id: 3, audioId: "z3", text: "Сірий, та не вовк,\nДовговухий, та не заєць,\nЗ копитами – та не кінь.", answerImg: "im3.png" },
    { id: 4, audioId: "z4", text: "Дуже товсті ноги маю,\nТихо їх переставляю.\nВисочезний я на зріст,\nБіля рота маю хвіст!", answerImg: "im4.png" },
    { id: 5, audioId: "z5", text: "Ця тварина каже: «Ме!\nЧи ви знаєте мене?»", answerImg: "im5.png" },
    { id: 6, audioId: "z6", text: "Біла латка, чорна латка\nПо березі скаче.", answerImg: "im6.png" },
    { id: 7, audioId: "z7", text: "Голодна – мукає,\nСита – жує,\nВсім дітям молока дає.", answerImg: "im7.png" },
    { id: 8, audioId: "z8", text: "На скелі він будує дім\nІ не боїться жити в нім.\nДва могутні є крила\nУ господаря …", answerImg: "im8.png" },
  ];

  // Масив зображень
  const imagesData = Array.from({length:8}, (_,i)=>({ img:`im${i+1}.png` }));

  const leftCol = document.getElementById("riddles-left");
  const rightCol = document.getElementById("riddles-right");
  const bankEl = document.getElementById("bank");
  const ok = document.getElementById("ok");
  const no = document.getElementById("no");
  const winMsg = document.getElementById("winMsg");

  const placed = new Map();
  const used = new Set();
  const shuffle = a => [...a].sort(()=>0.5-Math.random());

  function stopAll(){
    document.querySelectorAll("audio").forEach(a=>{ try{ a.pause(); a.currentTime=0; }catch(e){} });
  }
  function play(id){
    const a = document.getElementById(id);
    if(!a) return;
    stopAll();
    a.currentTime = 0;
    a.play().catch(()=>{});
  }

  document.addEventListener("click",(e)=>{
    const btn = e.target.closest("[data-a]");
    if(!btn) return;
    play(btn.dataset.a);
  });
  
  // Обробка кнопки перезапуску
  document.getElementById("btnRestart").addEventListener("click", init);

  function init(){
    placed.clear(); used.clear();
    leftCol.innerHTML = "";
    rightCol.innerHTML = "";
    bankEl.innerHTML = "";
    winMsg.style.display = "none"; // Приховуємо перемогу

    // 1. Розподіляємо загадки
    const shuffledRiddles = shuffle(riddlesData);
    
    shuffledRiddles.forEach((r, index) => {
      const card = document.createElement("div");
      card.className = "riddle";
      card.innerHTML = `
        <div class="rtop">
          <button class="snd" data-a="${r.audioId}" title="🔊">🔊</button>
          <div class="rtext">${esc(r.text)}</div>
        </div>
        <div class="drop" data-drop="${r.id}"></div>
      `;
      
      if (index < 4) {
        leftCol.appendChild(card);
      } else {
        rightCol.appendChild(card);
      }
    });

    // 2. Банк зображень
    shuffle(imagesData).forEach(it=>{
      const el = document.createElement("div");
      el.className = "item";
      el.draggable = true;
      el.dataset.img = it.img;
      el.innerHTML = `<img src="${it.img}" alt="">`;
      el.addEventListener("dragstart",(e)=>{
        if(el.classList.contains("used")) return e.preventDefault();
        e.dataTransfer.setData("text/plain", it.img);
      });
      bankEl.appendChild(el);
    });

    // 3. Dropzone
    document.querySelectorAll(".drop").forEach(zone=>{
      zone.addEventListener("dragover",(e)=>{ 
        e.preventDefault(); 
        if(zone.classList.contains("locked")) return; 
        zone.classList.add("over"); 
      });
      zone.addEventListener("dragleave",()=>zone.classList.remove("over"));
      zone.addEventListener("drop",(e)=>{
        e.preventDefault(); zone.classList.remove("over");
        if(zone.classList.contains("locked")) return;
        
        const img = e.dataTransfer.getData("text/plain");
        if(!img) return;
        
        const riddleId = Number(zone.dataset.drop);
        dropTo(riddleId, img, zone);
      });
    });

    // 4. Повернення в банк
    bankEl.addEventListener("dragover",(e)=>e.preventDefault());
    bankEl.addEventListener("drop",(e)=>{
      e.preventDefault();
      const img = e.dataTransfer.getData("text/plain");
      const from = e.dataTransfer.getData("fromRiddle");
      
      if(!img || !from) return; 
      
      const rid = Number(from);
      const zone = document.querySelector(`.drop[data-drop="${rid}"]`);
      const chip = zone?.querySelector(".chip");
      
      if(!zone || (chip && chip.classList.contains("ok"))) return; 
      
      zone.classList.remove("locked");
      zone.innerHTML = "";
      
      placed.delete(rid);
      used.delete(img);
      markUsed(img, false);
    });
  }

  function dropTo(riddleId, img, zone){
    if(used.has(img)) { pulse(zone); return; }

    const r = riddlesData.find(x=>x.id===riddleId);
    if(!r) return;

    if(placed.has(riddleId)){
      const prev = placed.get(riddleId);
      used.delete(prev);
      markUsed(prev, false);
    }

    zone.innerHTML = "";
    const chip = document.createElement("div");
    chip.className = "chip";
    chip.draggable = true;
    chip.innerHTML = `<img src="${img}" alt="">`;
    
    chip.addEventListener("dragstart",(e)=>{
      if(chip.classList.contains("ok")) return e.preventDefault();
      e.dataTransfer.setData("text/plain", img);
      e.dataTransfer.setData("fromRiddle", String(riddleId));
    });
    
    zone.appendChild(chip);

    placed.set(riddleId,img);
    used.add(img);
    markUsed(img, true);

    if(img === r.answerImg){
      chip.classList.add("ok");
      zone.classList.add("locked");
      ok.currentTime=0; ok.play().catch(()=>{});
      
      // Перевірка перемоги
      checkWin();
    }else{
      chip.classList.add("bad");
      no.currentTime=0; no.play().catch(()=>{});
      setTimeout(()=>chip.classList.remove("bad"), 550);
    }
  }

  function checkWin() {
    const solvedCount = document.querySelectorAll('.drop.locked').length;
    if (solvedCount === riddlesData.length) {
      setTimeout(() => {
        // Показуємо flex, щоб вирівняти кнопку і текст
        winMsg.style.display = "flex";
      }, 500);
    }
  }

  function markUsed(img, val){
    const el = [...document.querySelectorAll("#bank .item")].find(x=>x.dataset.img===img);
    if(!el) return;
    el.classList.toggle("used", !!val);
  }

  function pulse(el){
    el.style.transition="0.12s";
    el.style.transform="translateX(-6px)";
    el.style.borderColor="rgba(239,68,68,.75)";
    setTimeout(()=>el.style.transform="translateX(6px)",80);
    setTimeout(()=>{ el.style.transform="translateX(0)"; el.style.borderColor=""; },170);
  }

  const esc = s => String(s)
    .replaceAll("&","&amp;").replaceAll("<","&lt;").replaceAll(">","&gt;")
    .replaceAll('"',"&quot;").replaceAll("'","&#039;").replaceAll("\n","<br>");

  init();
</script>
