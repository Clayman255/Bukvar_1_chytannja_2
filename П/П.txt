<script>
  // Дані про картинки: 1-5 містять "п", 6 (коза) - ні.
  const itemsData = [
    { id: 1, img: '1.png', text: 'Порося', correct: true },
    { id: 2, img: '2.png', text: 'Папуга', correct: true },
    { id: 3, img: '3.png', text: 'Пиріжки', correct: true },
    { id: 4, img: '4.png', text: 'Півень', correct: true },
    { id: 5, img: '5.png', text: 'Павук', correct: true },
    { id: 6, img: '6.png', text: 'Коза', correct: false }
  ];

  const board = document.getElementById('board');
  const audioTask = document.getElementById('audioTask');
  const winMsg = document.getElementById('winMsg');
  const soundCorrect = document.getElementById('soundCorrect');
  // const soundWrong = document.getElementById('soundWrong'); 

  let correctCount = 0;
  // Рахуємо, скільки всього правильних відповідей у масиві
  const totalCorrect = itemsData.filter(i => i.correct).length;

  function initGame() {
    board.innerHTML = '';
    winMsg.style.display = 'none';
    correctCount = 0;

    // Перемішуємо масив і створюємо елементи
    [...itemsData].sort(() => 0.5 - Math.random()).forEach(item => {
      const el = document.createElement('div');
      el.className = 'item';
      
      // Створюємо HTML картки: картинка + текст
      el.innerHTML = `
        <img src="${item.img}" alt="${item.text}">
        <div class="item-label">${item.text}</div>
      `;

      // Додаємо обробку кліку
      el.addEventListener('click', () => handleItemClick(el, item));
      
      board.appendChild(el);
    });
  }

  function handleItemClick(element, itemData) {
    // Якщо вже відгадано (зелене), нічого не робимо
    if (element.classList.contains('success')) return;

    if (itemData.correct) {
      // Правильна відповідь
      element.classList.add('success');
      soundCorrect.currentTime = 0;
      soundCorrect.play().catch(() => {}); // Ігноруємо помилки автовідтворення
      
      correctCount++;
      
      // Перевірка перемоги
      if (correctCount === totalCorrect) {
        setTimeout(() => {
          winMsg.style.display = 'block';
        }, 300);
      }
    } else {
      // Неправильна відповідь
      element.classList.add('error');
      // soundWrong.play(); // Можна увімкнути
      
      // Прибираємо клас помилки після анімації, щоб можна було клікнути знову (хоча це помилка)
      setTimeout(() => {
        element.classList.remove('error');
      }, 500);
    }
  }

  // Кнопка звуку
  document.getElementById('btnSound').onclick = () => {
    audioTask.currentTime = 0;
    audioTask.play().catch(e => console.log("Audio play error", e));
  };

  // Кнопка рестарту
  document.getElementById('btnReset').onclick = initGame;

  // Запуск гри при завантаженні
  initGame();
</script>
