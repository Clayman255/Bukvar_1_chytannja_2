<script>
  const scene = document.getElementById("scene");
  const foundList = document.getElementById("foundList");
  const winMsg = document.getElementById("winMsg");
  const audioFile = document.getElementById("audioFile");
  const btnSound = document.getElementById("btnSound");
  
  const totalLetters = 10;
  let foundCount = 0;
  let positions = [];

  function initGame() {
    scene.innerHTML = "";
    foundList.innerHTML = "";
    winMsg.style.display = "none";
    foundCount = 0;
    positions = [];

    for (let i = 1; i <= 10; i++) {
      const li = document.createElement("li");
      li.className = "found-item";
      li.id = "item-" + i;
      li.innerHTML = `<span>${i}.</span> <span class="slot"></span>`;
      foundList.appendChild(li);
    }

    for (let i = 0; i < totalLetters; i++) {
      createLetter();
    }
  }

  function createLetter() {
    const letter = document.createElement("div");
    letter.className = "letter-c";
    letter.innerText = "С";
    
    let x, y, overlap;
    let attempts = 0;

    do {
      overlap = false;
      x = Math.random() * 85 + 5;
      y = Math.random() * 80 + 5;
      attempts++;

      for (let pos of positions) {
        let dist = Math.sqrt(Math.pow(x - pos.x, 2) + Math.pow(y - pos.y, 2));
        if (dist < 12) { overlap = true; break; }
      }
      if(attempts > 200) break; 
    } while (overlap);

    positions.push({x, y});
    letter.style.left = x + "%";
    letter.style.top = y + "%";

    letter.onclick = function() {
      handleFind(this);
    };

    scene.appendChild(letter);
  }

  function handleFind(el) {
    // Звук при натисканні на букви видалено за вашим запитом
    el.classList.add('found');

    setTimeout(() => {
      el.style.opacity = "0";
      el.style.pointerEvents = "none";
      
      foundCount++;
      const slot = document.querySelector(`#item-${foundCount} .slot`);
      if (slot) {
        slot.innerHTML = `<span class="collected-c">С</span>`;
      }

      if (foundCount === totalLetters) {
        setTimeout(() => {
          winMsg.style.display = "block";
        }, 500);
      }
    }, 2000); 
  }

  // Логіка кнопки звуку (Play/Pause)
  btnSound.onclick = () => {
    if (audioFile.paused) {
      audioFile.play();
      btnSound.innerText = "⏸";
    } else {
      audioFile.pause();
      btnSound.innerText = "🔊";
    }
  };

  // Повернення іконки після завершення звуку
  audioFile.onended = () => {
    btnSound.innerText = "🔊";
  };

  document.getElementById("btnReset").onclick = () => {
    // При скиданні гри звук також зупиняємо
    audioFile.pause();
    audioFile.currentTime = 0;
    btnSound.innerText = "🔊";
    initGame();
  };

  initGame();
</script>
