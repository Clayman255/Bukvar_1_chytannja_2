    <script>
        const data = [
            { id: 1, img: '1.png', imgAudio: '1.mp3', word: 'поїде', wordAudio: '11.mp3' },
            { id: 2, img: '2.png', imgAudio: '2.mp3', word: 'полетить', wordAudio: '22.mp3' },
            { id: 3, img: '3.png', imgAudio: '3.mp3', word: 'плив', wordAudio: '33.mp3' },
            { id: 4, img: '4.png', imgAudio: '4.mp3', word: 'їде', wordAudio: '44.mp3' }
        ];

        let selectedId = null;
        let selectedType = null;
        let connections = [];
        let matchedCount = 0;

        const canvas = document.getElementById('line-canvas');
        const ctx = canvas.getContext('2d');

        function playSound(file) {
            new Audio(file).play().catch(e => console.log("Файл не знайдено: " + file));
        }

        function initGame() {
            const leftImgCol = document.getElementById('left-img-col');
            const rightImgCol = document.getElementById('right-img-col');
            const wordCol = document.getElementById('word-col');

            data.forEach((item, index) => {
                const card = document.createElement('div');
                card.className = 'item-card';
                card.id = `img-${item.id}`;
                card.onclick = () => handleItemClick(item.id, 'img');
                card.innerHTML = `
                    <img src="${item.img}" class="animal-img">
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.imgAudio}')">🔊</button>
                `;
                if(index < 2) leftImgCol.appendChild(card);
                else rightImgCol.appendChild(card);
            });

            const shuffled = [...data].sort(() => Math.random() - 0.5);
            shuffled.forEach(item => {
                const card = document.createElement('div');
                card.className = 'item-card';
                card.id = `word-${item.id}`;
                card.onclick = () => handleItemClick(item.id, 'word');
                card.innerHTML = `
                    <div class="word-text">${item.word}</div>
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.wordAudio}')">🔊</button>
                `;
                wordCol.appendChild(card);
            });

            window.addEventListener('resize', resizeCanvas);
            setTimeout(resizeCanvas, 150);
        }

        function handleItemClick(id, type) {
            const currentEl = document.getElementById(`${type}-${id}`);
            if (currentEl.classList.contains('correct')) return;

            if (selectedId === id && selectedType === type) {
                clearSelection();
                return;
            }

            if (!selectedType || selectedType === type) {
                clearSelection();
                selectedId = id;
                selectedType = type;
                currentEl.classList.add('selected');
                const audio = type === 'img' ? data.find(d => d.id === id).imgAudio : data.find(d => d.id === id).wordAudio;
                playSound(audio);
            } else {
                if (selectedId === id) {
                    document.getElementById(`img-${id}`).classList.add('correct');
                    document.getElementById(`word-${id}`).classList.add('correct');
                    connections.push({ id });
                    matchedCount++;
                    clearSelection();
                    drawArrows();

                    if (matchedCount === data.length) {
                        document.getElementById('win-text').style.display = 'block';
                        playSound('win.mp3');
                    }
                } else {
                    clearSelection();
                    selectedId = id;
                    selectedType = type;
                    currentEl.classList.add('selected');
                    const audio = type === 'img' ? data.find(d => d.id === id).imgAudio : data.find(d => d.id === id).wordAudio;
                    playSound(audio);
                }
            }
        }

        function clearSelection() {
            document.querySelectorAll('.item-card').forEach(el => el.classList.remove('selected'));
            selectedId = null;
            selectedType = null;
        }

        function resizeCanvas() {
            canvas.width = document.documentElement.scrollWidth;
            canvas.height = document.documentElement.scrollHeight;
            drawArrows();
        }

        function drawArrows() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#3498db';
            ctx.lineWidth = 5;
            ctx.lineCap = 'round';

            connections.forEach(conn => {
                const imgEl = document.getElementById(`img-${conn.id}`);
                const wordEl = document.getElementById(`word-${conn.id}`);
                if(!imgEl || !wordEl) return;
                
                const imgRect = imgEl.getBoundingClientRect();
                const wordRect = wordEl.getBoundingClientRect();

                let startX;
                if (imgRect.left < wordRect.left) {
                    startX = imgRect.right + window.scrollX;
                } else {
                    startX = imgRect.left + window.scrollX;
                }

                const startY = imgRect.top + imgRect.height / 2 + window.scrollY;
                
                let endX;
                if (imgRect.left < wordRect.left) {
                    endX = wordRect.left + window.scrollX;
                } else {
                    endX = wordRect.right + window.scrollX;
                }
                
                const endY = wordRect.top + wordRect.height / 2 + window.scrollY;

                ctx.beginPath();
                ctx.moveTo(startX, startY);
                ctx.lineTo(endX, endY);
                ctx.stroke();

                ctx.fillStyle = '#3498db';
                ctx.beginPath();
                ctx.arc(startX, startY, 6, 0, Math.PI * 2);
                ctx.arc(endX, endY, 6, 0, Math.PI * 2);
                ctx.fill();
            });
        }

        window.onload = initGame;
    </script>
