<script>
const DATA = [
  { id: "a1", img: "1.png", foods: ["11.png"] },
  { id: "a2", img: "2.png", foods: ["22.png"] },
  { id: "a3", img: "3.png", foods: ["33.png"] },
  { id: "a4", img: "4.png", foods: ["44.png"] },
  { id: "a5", img: "5.png", foods: ["22.png"] },
  { id: "a6", img: "6.png", foods: ["11.png"] }
];

const els = {
  pool: document.getElementById('foodPool'),
  leftCol: document.getElementById('leftCol'),
  rightCol: document.getElementById('rightCol'),
  resetBtn: document.getElementById('resetBtn'),
  winMsg: document.getElementById('winMsg'),
  taskAudio: document.getElementById('taskAudio'),
  playBtn: document.getElementById('playTaskBtn')
};

let selectedFood = null;

// Відтворення звуку zav.mp3
els.playBtn.onclick = () => {
  els.taskAudio.currentTime = 0;
  els.taskAudio.play().catch(e => console.log("Audio play failed", e));
};

function initGame() {
  els.pool.innerHTML = '';
  els.leftCol.innerHTML = '';
  els.rightCol.innerHTML = '';
  els.winMsg.classList.remove('show');
  selectedFood = null;

  let allFoods = [];
  
  DATA.forEach((animal, index) => {
    const targetCol = (index < 3) ? els.leftCol : els.rightCol;
    createAnimalRow(animal, targetCol);
    
    animal.foods.forEach(fSrc => {
      allFoods.push({ src: fSrc });
    });
  });

  allFoods.sort(() => Math.random() - 0.5);
  allFoods.forEach(item => createFoodItem(item.src));
}

function createAnimalRow(animalData, targetEl) {
  const row = document.createElement('div');
  row.className = 'row';
  row.dataset.needed = JSON.stringify(animalData.foods);
  row.dataset.eaten = JSON.stringify([]);

  const container = document.createElement('div');
  container.className = 'animal-container';
  const img = document.createElement('img');
  img.src = animalData.img;
  img.className = 'adult-img';
  
  const check = document.createElement('div');
  check.className = 'check-mark';
  check.innerHTML = '✔';
  
  container.appendChild(img);
  container.appendChild(check);

  const slotsDiv = document.createElement('div');
  slotsDiv.className = 'slots-container';

  animalData.foods.forEach(() => {
    const slot = document.createElement('div');
    slot.className = 'slot';
    slot.onclick = () => handleSlotClick(slot, row);
    
    slot.ondragover = (e) => { e.preventDefault(); slot.style.borderColor = "var(--select)"; };
    slot.ondragleave = () => { slot.style.borderColor = ""; };
    slot.ondrop = (e) => {
      e.preventDefault();
      slot.style.borderColor = "";
      const foodSrc = e.dataTransfer.getData('text');
      const id = e.dataTransfer.getData('uid');
      checkMatch(id, foodSrc, row, slot);
    };

    slotsDiv.appendChild(slot);
  });

  row.appendChild(container);
  row.appendChild(slotsDiv);
  targetEl.appendChild(row);
}

function createFoodItem(src) {
  const img = document.createElement('img');
  img.src = src;
  img.className = 'food-item';
  img.draggable = true;
  const uid = 'f_' + Math.random().toString(36).substr(2, 9);
  img.id = uid; 
  img.dataset.src = src;

  img.onclick = (e) => { e.stopPropagation(); selectFood(img); };
  img.ondragstart = (e) => {
    selectFood(img);
    e.dataTransfer.setData('text', src);
    e.dataTransfer.setData('uid', uid);
  };

  els.pool.appendChild(img);
}

function selectFood(imgNode) {
  if (imgNode.parentElement.classList.contains('slot')) return;
  document.querySelectorAll('.food-item.selected').forEach(b => b.classList.remove('selected'));
  imgNode.classList.add('selected');
  selectedFood = imgNode;
}

function handleSlotClick(slot, row) {
  if (slot.children.length > 0 || row.classList.contains('completed') || !selectedFood) return;
  checkMatch(selectedFood.id, selectedFood.dataset.src, row, slot);
}

function checkMatch(itemUid, foodSrc, row, slot) {
  const foodEl = document.getElementById(itemUid);
  if (!foodEl) return;
  
  const needed = JSON.parse(row.dataset.needed);
  let eaten = JSON.parse(row.dataset.eaten);

  if (needed.includes(foodSrc) && eaten.filter(x => x === foodSrc).length < needed.filter(x => x === foodSrc).length) {
    slot.appendChild(foodEl);
    foodEl.draggable = false;
    foodEl.classList.remove('selected');
    selectedFood = null;

    eaten.push(foodSrc);
    row.dataset.eaten = JSON.stringify(eaten);

    if (eaten.length === needed.length) row.classList.add('completed');
    checkGlobalWin();
  } else {
    foodEl.classList.add('shake');
    setTimeout(() => foodEl.classList.remove('shake'), 400);
  }
}

function checkGlobalWin() {
  const allCompleted = document.querySelectorAll('.row.completed').length;
  if (allCompleted === DATA.length) {
    els.winMsg.classList.add('show');
  }
}

els.resetBtn.onclick = initGame;
initGame();
