<script>
    const quizData = [

            q: "Який це транспорт?",
            qAudio: "0.mp3",

            ]

        {
            q: "Який це транспорт?",
            qAudio: "0.mp3",
            img: "2.png",
            type: "text",
            options: [
                { text: "пасажирський", audio: "1.mp3", correct: true },
                { text: "вантажний", audio: "2.mp3", correct: false },
                { text: "морський", audio: "3.mp3", correct: false }
            ]
        },
        {
            q: "Який це транспорт?",
            qAudio: "0.mp3",
            img: "3.png",
            type: "text",
            options: [
                { text: "повітряний", audio: "4.mp3", correct: false },
                { text: "наземний", audio: "5.mp3", correct: false },
                { text: "морський", audio: "3.mp3", correct: true }
            ]
        },
        {
            q: "На змаганнях з мотокросу використовують …",
            qAudio: "6.mp3",
            type: "image",
            options: [
                { img: "4.png", correct: true },
                { img: "5.png", correct: false },
                { img: "6.png", correct: false }
            ]
        },
        {
            q: "У вітрильному спорті використовують …",
            qAudio: "7.mp3",
            type: "image",
            options: [
                { img: "7.png", correct: true },
                { img: "8.png", correct: false },
                { img: "9.png", correct: false }
            ]
        }
    ];

    let currentStep = 0;

    function playSound(file) {
        new Audio(file).play().catch(e => console.log("Файл не знайдено: " + file));
    }

    function loadQuestion() {
        const data = quizData[currentStep];
        document.getElementById('progress').textContent = `Питання ${currentStep + 1} з ${quizData.length}`;
        
        // Текст питання з іконкою звуку
        document.getElementById('q-text').innerHTML = `
            <button class="audio-btn" onclick="playSound('${data.qAudio}')">🔊</button>
            <span>${data.q}</span>
        `;

        // Малюнок до питання (якщо є)
        const imgContainer = document.getElementById('q-image-container');
        imgContainer.innerHTML = data.img ? `<img src="${data.img}" class="main-img">` : '';

        const optionsContainer = document.getElementById('options');
        optionsContainer.innerHTML = '';

        if (data.type === 'text') {
            optionsContainer.className = 'options-grid';
            data.options.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = 'option-btn';
                btn.innerHTML = `
                    <span>${opt.text}</span>
                    <div class="audio-btn" onclick="event.stopPropagation(); playSound('${opt.audio}')">🔊</div>
                `;
                btn.onclick = () => checkAnswer(opt.correct, btn);
                optionsContainer.appendChild(btn);
            });
        } else {
            optionsContainer.className = 'options-grid image-grid';
            data.options.forEach(opt => {
                const div = document.createElement('div');
                div.className = 'img-option';
                div.innerHTML = `<img src="${opt.img}">`;
                div.onclick = () => checkAnswer(opt.correct, div);
                optionsContainer.appendChild(div);
            });
        }
    }

    function checkAnswer(isCorrect, element) {
        if (isCorrect) {
            element.classList.add('correct');
            document.querySelectorAll('.option-btn, .img-option').forEach(el => el.style.pointerEvents = 'none');
            
            setTimeout(() => {
                currentStep++;
                if (currentStep < quizData.length) {
                    loadQuestion();
                } else {
                    showWin();
                }
            }, 1200);
        } else {
            element.classList.add('wrong');
            setTimeout(() => element.classList.remove('wrong'), 400);
        }
    }

    function showWin() {
        document.getElementById('quiz').style.display = 'none';
        document.getElementById('win-screen').style.display = 'block';
    }

    window.onload = loadQuestion;
</script>
