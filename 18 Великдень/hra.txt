<script>
  const wordsData = [
    { text: "писанка", isEaster: true },
    { text: "крашанка", isEaster: true },
    { text: "яйце", isEaster: true },
    { text: "паска", isEaster: true },
    { text: "свічка", isEaster: true },
    { text: "дряпанка", isEaster: true },
    { text: "дзвони", isEaster: true },
    { text: "ялинка", isEaster: false },
    { text: "коляда", isEaster: false }
  ];

  let correctFound = 0;
  const totalCorrect = wordsData.filter(w => w.isEaster).length;

  function initGame() {
    const grid = document.getElementById('grid');
    const shuffledWords = [...wordsData].sort(() => Math.random() - 0.5);

    shuffledWords.forEach(word => {
      const card = document.createElement('div');
      card.className = 'word-card';
      card.textContent = word.text;
      
      card.onclick = function() {
        if (card.classList.contains('correct')) return;

        if (word.isEaster) {
          card.classList.add('correct');
          correctFound++;
          checkWin();
        } else {
          // Додаємо підсвітку
          card.classList.add('wrong');
          // Прибираємо підсвітку через 600мс, повертаючи до початкового стану
          setTimeout(() => {
            card.classList.remove('wrong');
          }, 600);
        }
      };

      grid.appendChild(card);
    });
  }

  function checkWin() {
    if (correctFound === totalCorrect) {
      const winMsg = document.getElementById('winMsg');
      winMsg.style.display = 'block';
      setTimeout(() => {
        winMsg.style.display = 'none';
      }, 5000);
    }
  }

  initGame();
</script>
