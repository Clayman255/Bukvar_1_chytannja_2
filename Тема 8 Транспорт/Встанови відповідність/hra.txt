    <script>
        const data = [
            { id: 1, img: '1.png', word: 'батискаф', audio: '1.mp3' },
            { id: 2, img: '2.png', word: 'гелікоптер', audio: '2.mp3' },
            { id: 3, img: '3.png', word: 'комбайн', audio: '3.mp3' },
            { id: 4, img: '4.png', word: 'ракета', audio: '4.mp3' },
            { id: 5, img: '5.png', word: 'трамвай', audio: '5.mp3' },
            { id: 6, img: '6.png', word: 'човен', audio: '6.mp3' }
        ];

        let selectedId = null;
        let selectedType = null;
        let matchedCount = 0;
        let connections = [];

        const canvas = document.getElementById('line-canvas');
        const ctx = canvas.getContext('2d');
        const gameArea = document.getElementById('game-area');

        function playSound(file) {
            new Audio(file).play().catch(() => {});
        }

        window.onload = () => {
            initGame();
            setTimeout(resizeCanvas, 100);
        };

        window.addEventListener('resize', resizeCanvas);

        function resizeCanvas() {
            canvas.width = gameArea.offsetWidth;
            canvas.height = gameArea.offsetHeight;
            drawArrows();
        }

        function initGame() {
            const leftCol = document.getElementById('left-targets');
            const rightCol = document.getElementById('right-targets');
            const wordsCol = document.getElementById('words-container');

            data.forEach((item, index) => {
                const zone = document.createElement('div');
                zone.className = 'drop-zone';
                zone.id = `target-${item.id}`;
                zone.onclick = () => handleItemClick(item.id, 'target');
                zone.innerHTML = `<img src="${item.img}" class="item-img">`;
                if (index < 3) leftCol.appendChild(zone);
                else rightCol.appendChild(zone);
            });

            const shuffled = [...data].sort(() => Math.random() - 0.5);
            shuffled.forEach(item => {
                const card = document.createElement('div');
                card.className = 'word-card';
                card.id = `word-${item.id}`;
                card.onclick = () => handleItemClick(item.id, 'word');
                card.innerHTML = `
                    <span>${item.word}</span>
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.audio}')">🔊</button>
                `;
                wordsCol.appendChild(card);
            });
        }

        function handleItemClick(id, type) {
            const el = document.getElementById(`${type}-${id}`);
            if (el.classList.contains('matched') || el.classList.contains('correct')) return;

            if (selectedId === id && selectedType === type) {
                clearSelection();
                return;
            }

            if (!selectedType || selectedType === type) {
                clearSelection();
                selectedId = id;
                selectedType = type;
                el.classList.add('selected');
            } else {
                if (selectedId === id) {
                    const wordEl = document.getElementById(`word-${id}`);
                    const targetEl = document.getElementById(`target-${id}`);
                    
                    wordEl.classList.remove('selected');
                    wordEl.classList.add('matched');
                    targetEl.classList.add('correct');
                    
                    connections.push({ id: id });
                    matchedCount++;
                    clearSelection();
                    drawArrows();
                    
                    if (matchedCount === data.length) {
                        const winText = document.getElementById('win-text');
                        winText.style.display = 'block';
                        setTimeout(() => { winText.style.display = 'none'; }, 5000);
                    }
                } else {
                    clearSelection();
                    selectedId = id;
                    selectedType = type;
                    el.classList.add('selected');
                }
            }
        }

        function clearSelection() {
            document.querySelectorAll('.word-card, .drop-zone').forEach(el => el.classList.remove('selected'));
            selectedId = null;
            selectedType = null;
        }

        function drawArrows() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#3498db';
            ctx.lineWidth = 4;
            ctx.lineCap = 'round';

            const areaRect = gameArea.getBoundingClientRect();

            connections.forEach(conn => {
                const wEl = document.getElementById(`word-${conn.id}`);
                const tEl = document.getElementById(`target-${conn.id}`);
                
                const wRect = wEl.getBoundingClientRect();
                const tRect = tEl.getBoundingClientRect();

                // Координати відносно контейнера game-layout
                const wordCenterY = (wRect.top + wRect.height / 2) - areaRect.top;
                const targetCenterY = (tRect.top + tRect.height / 2) - areaRect.top;

                let wordX, targetX;

                if (tRect.left < wRect.left) {
                    targetX = tRect.right - areaRect.left;
                    wordX = wRect.left - areaRect.left;
                } else {
                    targetX = tRect.left - areaRect.left;
                    wordX = wRect.right - areaRect.left;
                }

                ctx.beginPath();
                ctx.moveTo(targetX, targetCenterY);
                ctx.lineTo(wordX, wordCenterY);
                ctx.stroke();

                ctx.fillStyle = '#3498db';
                ctx.beginPath();
                ctx.arc(targetX, targetCenterY, 5, 0, Math.PI * 2);
                ctx.arc(wordX, wordCenterY, 5, 0, Math.PI * 2);
                ctx.fill();
            });
        }
