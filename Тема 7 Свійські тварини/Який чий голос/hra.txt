    <script>
        const data = [
            { id: 1, img: '1.png', sound: '11.mp3' },
            { id: 2, img: '2.png', sound: '22.mp3' },
            { id: 3, img: '3.png', sound: '33.mp3' },
            { id: 4, img: '4.png', sound: '44.mp3' }
        ];

        let selectedId = null;
        let selectedType = null;
        let matchedCount = 0;
        let connections = [];

        const canvas = document.getElementById('line-canvas');
        const ctx = canvas.getContext('2d');

        function playSound(file) {
            const audio = new Audio(file);
            audio.play().catch(e => console.log("Файл не знайдено: " + file));
        }

        window.onload = () => {
            initGame();
            resizeCanvas();
        };

        window.onresize = resizeCanvas;
        window.onscroll = drawArrows;

        function resizeCanvas() {
            canvas.width = document.documentElement.scrollWidth;
            canvas.height = document.documentElement.scrollHeight;
            drawArrows();
        }

        function initGame() {
            const leftCol = document.getElementById('left-targets');
            const rightCol = document.getElementById('right-targets');
            const soundsCol = document.getElementById('sounds-container');

            data.forEach((item, index) => {
                const zone = document.createElement('div');
                zone.className = 'drop-zone';
                zone.id = `target-${item.id}`;
                zone.onclick = () => handleItemClick(item.id, 'target');
                zone.innerHTML = `<img src="${item.img}" class="animal-img" alt="тварина">`;
                
                if (index < 2) leftCol.appendChild(zone);
                else rightCol.appendChild(zone);
            });

            const shuffled = [...data].sort(() => Math.random() - 0.5);
            shuffled.forEach(item => {
                const card = document.createElement('div');
                card.className = 'sound-card';
                card.id = `sound-${item.id}`;
                card.onclick = () => handleItemClick(item.id, 'sound');
                card.innerHTML = `
                    <button class="audio-btn" onclick="event.stopPropagation(); playSound('${item.sound}')">🔊</button>
                    <span>Голос</span>
                `;
                soundsCol.appendChild(card);
            });
        }

        function handleItemClick(id, type) {
            const el = document.getElementById(`${type}-${id}`);
            if (el.classList.contains('matched') || el.classList.contains('correct')) return;

            if (selectedId === id && selectedType === type) {
                clearSelection();
                return;
            }

            if (!selectedType || selectedType === type) {
                clearSelection();
                selectedId = id;
                selectedType = type;
                el.classList.add('selected');
                if (type === 'sound') playSound(data.find(d => d.id === id).sound);
            } 
            else {
                if (selectedId === id) {
                    const soundEl = document.getElementById(`sound-${id}`);
                    const targetEl = document.getElementById(`target-${id}`);
                    
                    soundEl.classList.remove('selected');
                    soundEl.classList.add('matched');
                    targetEl.classList.remove('selected');
                    targetEl.classList.add('correct');
                    
                    connections.push({ soundId: id, targetId: id });
                    matchedCount++;
                    clearSelection();
                    drawArrows();

                    if (matchedCount === data.length) {
                        document.getElementById('win-text').style.display = 'block';
                        setTimeout(() => playSound('win.mp3'), 500);
                    }
                } else {
                    clearSelection();
                    selectedId = id;
                    selectedType = type;
                    el.classList.add('selected');
                    if (type === 'sound') playSound(data.find(d => d.id === id).sound);
                }
            }
        }

        function clearSelection() {
            document.querySelectorAll('.sound-card, .drop-zone').forEach(el => el.classList.remove('selected'));
            selectedId = null;
            selectedType = null;
        }

        function drawArrows() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#3498db';
            ctx.lineWidth = 6;
            ctx.lineCap = 'round';

            connections.forEach(conn => {
                const sEl = document.getElementById(`sound-${conn.soundId}`);
                const tEl = document.getElementById(`target-${conn.targetId}`);
                
                const sRect = sEl.getBoundingClientRect();
                const tRect = tEl.getBoundingClientRect();

                let soundX;
                const soundCenterY = sRect.top + sRect.height / 2 + window.scrollY;
                let targetX;
                const targetCenterY = tRect.top + tRect.height / 2 + window.scrollY;

                if (tRect.left < sRect.left) {
                    targetX = tRect.right + window.scrollX;
                    soundX = sRect.left + window.scrollX;
                } else {
                    targetX = tRect.left + window.scrollX;
                    soundX = sRect.right + window.scrollX;
                }

                ctx.beginPath();
                ctx.moveTo(targetX, targetCenterY);
                ctx.lineTo(soundX, soundCenterY);
                ctx.stroke();

                ctx.fillStyle = '#3498db';
                ctx.beginPath();
                ctx.arc(targetX, targetCenterY, 8, 0, Math.PI * 2);
                ctx.arc(soundX, soundCenterY, 8, 0, Math.PI * 2);
                ctx.fill();
            });
        }
