<script>
const riddles = [
  {
    id:1,
    audio:"au1",
    text:
`На однім кінці – дивак,
А на другому – черв’як.
Увесь день на черв’яка
Всі надії в дивака.`,
    img:"im1.png"
  },
  {
    id:2,
    audio:"au2",
    text:
`На собі я ношу дім,
Від хижаків ховаюсь в нім.`,
    img:"im2.png"
  },
  {
    id:3,
    audio:"au3",
    text:
`Живий – чорний,
А як звариться – червоний.`,
    img:"im3.png"
  },
  {
    id:4,
    audio:"au4",
    text:
`Крила має, та не літає,
Ніг не має, а гуляє,
По землі не ходить
І людей не знає.`,
    img:"im4.png"
  }
];

const left = document.getElementById("left");
const right = document.getElementById("right");
const bank = document.getElementById("bank");
const win = document.getElementById("win");

let placed = new Map();
let used = new Set();

riddles.forEach((r,i)=>{
  const card = document.createElement("div");
  card.className="riddle";
  card.innerHTML=`
    <div class="rtop">
      <button class="snd" data-a="${r.audio}">🔊</button>
      <div class="rtext">${r.text.replace(/\n/g,"<br>")}</div>
    </div>
    <div class="drop" data-id="${r.id}"></div>
  `;
  (i<2?left:right).appendChild(card);
});

["im1.png","im2.png","im3.png","im4.png"].forEach(src=>{
  const el=document.createElement("div");
  el.className="item";
  el.draggable=true;
  el.dataset.img=src;
  el.innerHTML=`<img src="${src}">`;
  el.ondragstart=e=>e.dataTransfer.setData("img",src);
  bank.appendChild(el);
});

document.querySelectorAll(".drop").forEach(zone=>{
  zone.ondragover=e=>e.preventDefault();
  zone.ondrop=e=>{
    e.preventDefault();
    if(zone.classList.contains("locked")) return;
    const img=e.dataTransfer.getData("img");
    const id=+zone.dataset.id;
    const r=riddles.find(x=>x.id===id);
    if(!img||used.has(img)) return;

    zone.innerHTML=`<div class="chip"><img src="${img}"></div>`;
    used.add(img);
    document.querySelector(`.item[data-img="${img}"]`).classList.add("used");

    if(img===r.img){
      zone.classList.add("locked");
      placed.set(id,img);
      if(placed.size===4) win.style.display="flex";
    }
  };
});

document.addEventListener("click",e=>{
  const b=e.target.closest("[data-a]");
  if(!b) return;
  document.querySelectorAll("audio").forEach(a=>{a.pause();a.currentTime=0;});
  document.getElementById(b.dataset.a).play();
});
</script>
